162 
163  // set constraints (first hanging nodes, then flux)
164  constraints.clear();
165 
166  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
167 
168  /*
169  * Set true for Neumann BCs (hard constraints on sigma).
170  */
171  if (parameters.is_pure_neumann)
172  {
173  EquationData::Boundary_A_grad_u tensor_boundary_A_grad_u;
174  VectorFunctionFromTensorFunction<3> boundary_A_grad_u(
175  tensor_boundary_A_grad_u);
176 
177  for (unsigned int i = 0; i < GeometryInfo<3>::faces_per_cell; ++i)
178  VectorTools::project_boundary_values_div_conforming(
179  dof_handler,
180  /*first vector component */ 0,
181  boundary_A_grad_u,
182  /*boundary id*/ i,
183  constraints);
184  }
185 
186  /*
187  * If we have a Laplace problem (not Helmholtz) and a pure
188  * Neumann problem then we need to make sure that u is unique.
189  * We therefore add a constraint on dofs
190  */
191  if (parameters.is_laplace && parameters.is_pure_neumann)
192  {
193  IndexSet locally_relevant_dofs_u(relevant_partitioning[1]);
194 
195  // initially set a non-admissible value
196  unsigned int first_local_dof_u = dof_handler.n_dofs() + 1;
197  if (locally_relevant_dofs_u.n_elements() > 0)
198  first_local_dof_u = locally_relevant_dofs_u.nth_index_in_set(0);

333  RTDQMultiscale::solve_iterative()
334  {
335  inner_schur_preconditioner =
336  std::make_shared<typename LinearSolvers::InnerPreconditioner<3>::type>();
337 
338  typename LinearSolvers::InnerPreconditioner<3>::type::AdditionalData data;
339 
340  inner_schur_preconditioner->initialize(system_matrix.block(0, 0), data);
341 
342  /*
343  * Define the inverse of the first
344  * block by its action.
345  */
346  const LinearSolvers::InverseMatrix<
347  LA::MPI::SparseMatrix,
348  typename LinearSolvers::InnerPreconditioner<3>::type>
349  block_inverse(system_matrix.block(0, 0), *inner_schur_preconditioner);
350 
351  // Vector for solution
352  LA::MPI::BlockVector distributed_solution(owned_partitioning,
353  mpi_communicator);
354 
355  // tmp of size block(0)
356  LA::MPI::Vector tmp(owned_partitioning[0], mpi_communicator);
357 
358  // Set up Schur complement
359  LinearSolvers::SchurComplementMPI<
360  LA::MPI::BlockSparseMatrix,
361  LA::MPI::Vector,
362  typename LinearSolvers::InnerPreconditioner<3>::type>
363  schur_complement(system_matrix,
364  block_inverse,
365  owned_partitioning,
366  mpi_communicator);
367 
368  // Compute schur_rhs = -g + C*A^{-1}*f
369  LA::MPI::Vector schur_rhs(owned_partitioning[1],
370  // relevant_partitioning[1],
371  mpi_communicator);
372  block_inverse.vmult(tmp, system_rhs.block(0));
373  system_matrix.block(1, 0).vmult(schur_rhs, tmp);
374  schur_rhs -= system_rhs.block(1);
375  {
376  TimerOutput::Scope t(computing_timer, "Schur complement solver (for u)");
377 
378  // Set Solver parameters for solving for u
379  SolverControl solver_control(system_matrix.m(),
380  1e-6 * schur_rhs.l2_norm());
381  SolverCG<LA::MPI::Vector> schur_solver(solver_control);
382 
384  /*
385  * Use no preconditioner
386  */
387  // PreconditionIdentity preconditioner;
388 
389  /*
390  * Precondition the Schur complement with
391  * the approximate inverse of the
392  * Schur complement.
393  */
394  // LinearSolvers::ApproximateInverseMatrix<LinearSolvers::SchurComplementMPI<LA::MPI::BlockSparseMatrix,
395  // LA::MPI::Vector,
396  // typename
397  // LinearSolvers::InnerPreconditioner<3>::type>,
398  // PreconditionIdentity>
399  // preconditioner (schur_complement,
400  // PreconditionIdentity()
401  //);
402 
403  /*
404  * Precondition the Schur complement with
405  * the approximate inverse of an approximate
406  * Schur complement.
407  */
408  LinearSolvers::ApproximateSchurComplementMPI<LA::MPI::BlockSparseMatrix,
409  LA::MPI::Vector,
410  LA::MPI::PreconditionILU>
411  approx_schur(system_matrix, owned_partitioning, mpi_communicator);
412 
413  LinearSolvers::ApproximateInverseMatrix<
414  LinearSolvers::ApproximateSchurComplementMPI<LA::MPI::BlockSparseMatrix,
415  LA::MPI::Vector,
416  LA::MPI::PreconditionILU>,
417  PreconditionIdentity>
418  preconditioner(approx_schur,
419  PreconditionIdentity(),
420 #ifdef DEBUG
421  /* n_iter */ 1000);
422 #else
423  /* n_iter */ 20);
424 #endif
425 
426  /*
427  * Precondition the Schur complement with a preconditioner of
428  * block(1,1).
429  */
430  // LA::MPI::PreconditionAMG preconditioner;
431  // preconditioner.initialize(system_matrix.block(1, 1),
432  // data);
433 
435 
436  schur_solver.solve(schur_complement,
437  distributed_solution.block(1),
438  schur_rhs,
439  preconditioner);
440 
441  pcout << " Iterative Schur complement solver converged in "
442  << solver_control.last_step() << " iterations." << std::endl;
443 
444  constraints.distribute(distributed_solution);
445  }
446 
447  {
448  TimerOutput::Scope t(computing_timer, "outer CG solver (for sigma)");
449 
450  // SolverControl outer_solver_control;
451  // PETScWrappers::SparseDirectMUMPS
452  // outer_solver(outer_solver_control,
453  // mpi_communicator); outer_solver.set_symmetric_mode(true);
454 
455  // use computed u to solve for sigma
456  system_matrix.block(0, 1).vmult(tmp, distributed_solution.block(1));
457  tmp *= -1;
458  tmp += system_rhs.block(0);
459 
460  // Solve for sigma
461  // outer_solver.solve(system_matrix.block(0,0),
462  // distributed_solution.block(0), tmp);
463  block_inverse.vmult(distributed_solution.block(0), tmp);
464 
465  pcout << " Outer solver completed." << std::endl;
