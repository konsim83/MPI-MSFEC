167  * Matrix coefficient and vector to store the values.

171 
172  /*
173  * Reaction rate.
174  */
175  const EquationData::ReactionRate reaction_rate;
176  std::vector<double> reaction_rate_values(n_q_points);
177 
178  /*
179  * Right hand side and vector to store the values.
180  */

182  parameter_filename, /* n_components */ 1);

184 
185  /*
186  * Neumann BCs and vector to store the values.
187  */
188  const EquationData::Boundary_A_grad_u neumann_bc;
189  std::vector<Tensor<1, 3>> neumann_values(n_face_q_points);
190 
191  /*
192  * Integration over cells.
193  */
194  for (const auto &cell : dof_handler.active_cell_iterators())
195  {
196  if (cell->is_locally_owned())
197  {
198  cell_matrix = 0;
199  cell_rhs = 0;
200 
201  fe_values.reinit(cell);
202 
203  // Now actually fill with values.
204  matrix_coeff.value_list(fe_values.get_quadrature_points(),
205  matrix_coeff_values);
206  reaction_rate.value_list(fe_values.get_quadrature_points(),
207  reaction_rate_values);
208  right_hand_side.value_list(fe_values.get_quadrature_points(),
209  rhs_values);
210 
211  for (unsigned int q_index = 0; q_index < n_q_points; ++q_index)
212  {
213  for (unsigned int i = 0; i < dofs_per_cell; ++i)
214  {
215  for (unsigned int j = 0; j < dofs_per_cell; ++j)
216  {
217  cell_matrix(i, j) +=
218  (fe_values.shape_grad(i, q_index) *
219  matrix_coeff_values[q_index] *
220  fe_values.shape_grad(j, q_index) +
221  fe_values.shape_value(i, q_index) *
222  reaction_rate_values[q_index] *
223  fe_values.shape_value(j, q_index)) *
224  fe_values.JxW(q_index);
225  } // end ++j
226 
227  cell_rhs(i) += fe_values.shape_value(i, q_index) *
228  rhs_values[q_index] * fe_values.JxW(q_index);
229  } // end ++i
230  } // end ++q_index
231 
232  if (parameters.is_pure_neumann)
233  {
234  /*
235  * Boundary integral for Neumann values for odd boundary_id.
236  */
237  for (unsigned int face_number = 0;
238  face_number < GeometryInfo<3>::faces_per_cell;
239  ++face_number)
240  {
241  if (cell->face(face_number)->at_boundary())
242  {
243  fe_face_values.reinit(cell, face_number);
244 
245  /*
246  * Fill in values at this particular face.
247  */
248  neumann_bc.value_list(
249  fe_face_values.get_quadrature_points(),
250  neumann_values);
251 
252  for (unsigned int q_face_point = 0;
253  q_face_point < n_face_q_points;
254  ++q_face_point)
255  {
256  for (unsigned int i = 0; i < dofs_per_cell; ++i)
257  {
258  cell_rhs(i) +=
259  (neumann_values[q_face_point] *
260  fe_face_values.normal_vector(
261  q_face_point) // g(x_q)
262  ) *
263  fe_face_values.shape_value(
264  i, q_face_point) // phi_i(x_q)
265  * fe_face_values.JxW(q_face_point); // dS
266  } // end ++i
267  } // end ++q_face_point

269  } // end ++face_number
270  }
271

278  * be taken care of later.

280  constraints.distribute_local_to_global(cell_matrix,
