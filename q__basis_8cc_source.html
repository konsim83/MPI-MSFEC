175  QBasis::assemble_system()
176  {
177  QGauss<3> quadrature_formula(fe.degree + 1);
178 
179  FEValues<3> fe_values(fe,
180  quadrature_formula,
181  update_values | update_gradients |
182  update_quadrature_points | update_JxW_values);
183 
184  const unsigned int dofs_per_cell = fe.dofs_per_cell;
185  const unsigned int n_q_points = quadrature_formula.size();
186 
187  FullMatrix<double> cell_diffusion_matrix(dofs_per_cell, dofs_per_cell);
188  FullMatrix<double> cell_mass_matrix(dofs_per_cell, dofs_per_cell);
189  Vector<double> cell_rhs(dofs_per_cell);
190 
191  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
192 
193  /*
194  * Matrix coefficient and vector to store the values.
195  */
196  const EquationData::Diffusion_A matrix_coeff(parameter_filename);
197  std::vector<Tensor<2, 3>> matrix_coeff_values(n_q_points);
198 
199  /*
200  * Right hand side and vector to store the values.
201  */
202  const EquationData::RightHandSideParsed right_hand_side(
203  parameter_filename, /* n_components */ 1);
204  std::vector<double> rhs_values(n_q_points);
205 
206  /*
207  * Integration over cells.
208  */
209  for (const auto &cell : dof_handler.active_cell_iterators())
210  {
211  cell_diffusion_matrix = 0;
212  cell_rhs = 0;
213 
214  fe_values.reinit(cell);
215 
216  // Now actually fill with values.
217  matrix_coeff.value_list(fe_values.get_quadrature_points(),
218  matrix_coeff_values);
219  right_hand_side.value_list(fe_values.get_quadrature_points(),
220  rhs_values);
221 
222  for (unsigned int q_index = 0; q_index < n_q_points; ++q_index)
223  {
224  for (unsigned int i = 0; i < dofs_per_cell; ++i)
225  {
226  for (unsigned int j = 0; j < dofs_per_cell; ++j)
227  {
228  cell_diffusion_matrix(i, j) +=
229  fe_values.shape_grad(i, q_index) *
230  matrix_coeff_values[q_index] *
231  fe_values.shape_grad(j, q_index) * fe_values.JxW(q_index);
232  } // end ++j
233 
234  cell_rhs(i) += fe_values.shape_value(i, q_index) *
235  rhs_values[q_index] * fe_values.JxW(q_index);
236  } // end ++i
237  } // end ++q_index
238 
239  // get global indices
240  cell->get_dof_indices(local_dof_indices);
241  /*
242  * Now add the cell matrix and rhs to the right spots
243  * in the global matrix and global rhs. Constraints will
244  * be taken care of later.
245  */
246  for (unsigned int i = 0; i < dofs_per_cell; ++i)
247  {
248  for (unsigned int j = 0; j < dofs_per_cell; ++j)
249  {
250  diffusion_matrix.add(local_dof_indices[i],
251  local_dof_indices[j],
252  cell_diffusion_matrix(i, j));
253  }
254  global_rhs(local_dof_indices[i]) += cell_rhs(i);
255  }
256  } // end ++cell
257  }

262  QBasis::assemble_global_element_matrix()
263  {
264  // First, reset.
265  global_element_matrix = 0;
266 
267  // Get lengths of tmp vectors for assembly
268  const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
269 
270  Vector<double> tmp(dof_handler.n_dofs());
271 
272  // This assembles the local contribution to the global global matrix
273  // with an algebraic trick. It uses the local system matrix stored in
274  // the respective basis object.
275  for (unsigned int i_test = 0; i_test < dofs_per_cell; ++i_test)
276  {
277  // set an alias name
278  const Vector<double> &test_vec = solution_vector[i_test];
279 
280  for (unsigned int i_trial = 0; i_trial < dofs_per_cell; ++i_trial)
281  {
282  // set an alias name
283  const Vector<double> &trial_vec = solution_vector[i_trial];
284 
285  // tmp = system_matrix*trial_vec
286  diffusion_matrix.vmult(tmp, trial_vec);
287 
288  // global_element_diffusion_matrix = test_vec*tmp
289  global_element_matrix(i_test, i_trial) += (test_vec * tmp);
290 
291  // reset
292  tmp = 0;
293  } // end for i_trial
294 
295  global_element_rhs(i_test) += test_vec * global_rhs;
296 
297  } // end for i_test
298 
299  is_built_global_element_matrix = true;
