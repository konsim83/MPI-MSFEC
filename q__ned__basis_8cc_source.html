385  Vector<double> local_rhs(dofs_per_cell);

387  GeometryInfo<3>::lines_per_cell + GeometryInfo<3>::vertices_per_cell,
388  Vector<double>(dofs_per_cell));

394  if (parameters.use_exact_solution)

396  new EquationData::RightHandSideExactLin(parameter_filename));
397  else
398  right_hand_side.reset(
399  new EquationData::RightHandSideParsed(parameter_filename,
400  /* n_components */ 3));
401 
402  const EquationData::Diffusion_A diffusion_a(parameter_filename);
403  const EquationData::DiffusionInverse_B diffusion_inverse_b(
404  parameter_filename, parameters.use_exact_solution);

409  std::vector<Tensor<2, 3>> diffusion_a_values(n_q_points);
410  std::vector<double> diffusion_inverse_b_values(n_q_points);
411  std::vector<double> reaction_rate_values(n_q_points);
412 
414  ShapeFun::BasisQ1Grad<3> std_shape_function_h1_grad(global_cell_it);

416  GeometryInfo<3>::lines_per_cell, std::vector<Tensor<1, 3>>(n_q_points));

420  const FEValuesExtractors::Vector curl(/* first_vector_component */ 1);
421

423  // loop over cells
424  typename DoFHandler<3>::active_cell_iterator cell =
425  dof_handler.begin_active(),
426  endc = dof_handler.end();
427  for (; cell != endc; ++cell)
428  {
429  fe_values.reinit(cell);
430

432  local_rhs = 0;
433 
434  for (unsigned int n_basis = 0; n_basis < length_system_basis; ++n_basis)

437

439  (parameters.full_rhs))

504  v_i * local_rhs_values[n_basis][q] * fe_values.JxW(q);
505  }
506  } // end for ++i
507  } // end for ++q
508

533  {

535  local_dof_indices[j],
536  local_matrix(i, j));

559  } // end for ++cell

565  << std::endl;

627  void
628  QNedBasis::solve_iterative(unsigned int n_basis)
629  {
630  Timer timer;
631  Timer inner_timer;
632 
633  // ------------------------------------------
634  // Make a preconditioner for each system matrix
635  if (parameters.verbose)
636  {
637  std::cout << " Computing preconditioner in cell "
638  << global_cell_id.to_string() << " for basis " << n_basis
639  << " .....";
640 
641  timer.restart();
642  }
643 
644  BlockVector<double> *system_rhs_ptr = NULL;
645  BlockVector<double> *solution_ptr = NULL;
646  if (n_basis < GeometryInfo<3>::vertices_per_cell)
647  {
648  system_rhs_ptr = &(system_rhs_h1_v[n_basis]);
649  solution_ptr = &(basis_h1_v[n_basis]);
650  }
651  else
652  {
653  const unsigned int offset_index =
654  n_basis - GeometryInfo<3>::vertices_per_cell;
655  system_rhs_ptr = &(system_rhs_curl_v[offset_index]);
656  solution_ptr = &(basis_curl_v[offset_index]);
657  }
658 
659  // for convenience
660  const BlockVector<double> &system_rhs = *system_rhs_ptr;
661  BlockVector<double> & solution = *solution_ptr;
662 
663  inner_schur_preconditioner = std::make_shared<
664  typename LinearSolvers::LocalInnerPreconditioner<3>::type>();
665 
666  typename LinearSolvers::LocalInnerPreconditioner<3>::type::AdditionalData
667  data;
668  inner_schur_preconditioner->initialize(system_matrix.block(0, 0), data);
669 
670  if (parameters.verbose)
671  {
672  timer.stop();
673  std::cout << "done in " << timer.cpu_time() << " seconds."
674  << std::endl;
675  }
676  // ------------------------------------------
677 
678  // Now solve.
679  if (parameters.verbose)
680  {
681  std::cout << " Solving linear system (iteratively, with "
682  "preconditioner) in cell "
683  << global_cell_id.to_string() << " for basis " << n_basis
684  << " .....";
685 
686  timer.restart();
687  }
688 
689  // Construct inverse of upper left block
690  const LinearSolvers::InverseMatrix<
691  SparseMatrix<double>,
692  typename LinearSolvers::LocalInnerPreconditioner<3>::type>
693  block_inverse(system_matrix.block(0, 0), *inner_schur_preconditioner);
694 
695  Vector<double> tmp(system_rhs.block(0).size());
696  {
697  // Set up Schur complement
698  LinearSolvers::SchurComplement<
699  BlockSparseMatrix<double>,
700  Vector<double>,
701  typename LinearSolvers::LocalInnerPreconditioner<3>::type>
702  schur_complement(system_matrix, block_inverse);
703 
704  // Compute schur_rhs = -g + C*A^{-1}*f
705  Vector<double> schur_rhs(system_rhs.block(1).size());
706 
707  block_inverse.vmult(tmp, system_rhs.block(0));
708  system_matrix.block(1, 0).vmult(schur_rhs, tmp);
709  schur_rhs -= system_rhs.block(1);
710 
711  {
712  if (parameters.verbose)
713  {
714  inner_timer.restart();
715  }
716  SolverControl solver_control(system_matrix.m(),
717  1e-6 * schur_rhs.l2_norm());
718  SolverCG<Vector<double>> schur_solver(solver_control);
719 
720  // PreconditionIdentity preconditioner;
721 
722  /*
723  * Precondition the Schur complement with
724  * the approximate inverse of the
725  * Schur complement.
726  */
727  // LinearSolvers::ApproximateInverseMatrix<LinearSolvers::SchurComplement<BlockSparseMatrix<double>,
728  // Vector<double>,
729  // typename
730  // LinearSolvers::LocalInnerPreconditioner<3>::type>,
731  // PreconditionIdentity>
732  // preconditioner
733  //(schur_complement, PreconditionIdentity(),
734  // /* n_iter */
735  // 5);
736 
737  /*
738  * Precondition the Schur complement with
739  * the (approximate) inverse of an approximate
740  * Schur complement.
741  */
742  // using ApproxSchurPrecon =
743  // PreconditionJacobi<SparseMatrix<double>>;
744  // using ApproxSchurPrecon =
745  // PreconditionSOR<SparseMatrix<double>>;
746  using ApproxSchurPrecon = SparseILU<double>;
747  // using ApproxSchurPrecon = PreconditionIdentity;
748  LinearSolvers::ApproximateSchurComplement<BlockSparseMatrix<double>,
749  Vector<double>,
750  ApproxSchurPrecon>
751  approx_schur(system_matrix);
752 
753  LinearSolvers::ApproximateInverseMatrix<
754  LinearSolvers::ApproximateSchurComplement<BlockSparseMatrix<double>,
755  Vector<double>,
756  ApproxSchurPrecon>,
757  PreconditionIdentity>
758  preconditioner(approx_schur,
759  PreconditionIdentity(),
760 #ifdef DEBUG
761  /* n_iter */ 1000);
762 #else
763  /* n_iter */ 14);
764 #endif
765 
766  schur_solver.solve(schur_complement,
767  solution.block(1),
768  schur_rhs,
769  preconditioner);
770 
771  if (n_basis < GeometryInfo<3>::vertices_per_cell)
772  {
773  constraints_h1_v[n_basis].distribute(solution);
774  }
775  else
776  {
777  const unsigned int offset_index =
778  n_basis - GeometryInfo<3>::vertices_per_cell;
779 
780  constraints_curl_v[offset_index].distribute(solution);
781  }
782 
783  if (parameters.verbose)
784  {
785  inner_timer.stop();
786 
787  std::cout << std::endl
788  << " - Iterative Schur complement solver "
789  "converged in "
790  << solver_control.last_step()
791  << " iterations. Time: " << inner_timer.cpu_time()
792  << " seconds." << std::endl;
793  }
794  }
795 
796  {
797  if (parameters.verbose)
798  {
799  inner_timer.restart();
800  }
801 
802  // use computed u to solve for sigma
803  system_matrix.block(0, 1).vmult(tmp, solution.block(1));
804  tmp *= -1;
805  tmp += system_rhs.block(0);
806 
807  // Solve for sigma
808  block_inverse.vmult(solution.block(0), tmp);
809 
810  if (parameters.verbose)
811  {
812  inner_timer.stop();
813 
814  std::cout << " - Outer solver completed. Time: "
815  << inner_timer.cpu_time() << " seconds." << std::endl;
816  }
817  }
818 
819  if (n_basis < GeometryInfo<3>::vertices_per_cell)
820  {
821  constraints_h1_v[n_basis].distribute(solution);
822  }
823  else
824  {
825  const unsigned int offset_index =
826  n_basis - GeometryInfo<3>::vertices_per_cell;
827 
828  constraints_curl_v[offset_index].distribute(solution);
829  }
830  }
831 
832  if (parameters.verbose)
833  {
834  timer.stop();
835  std::cout << " - done in " << timer.cpu_time() << " seconds."
836  << std::endl;
837  }

840  void
841  QNedBasis::assemble_global_element_matrix()
842  {
843  // First, reset.
844  global_element_matrix = 0;
845 
846  // Get lengths of tmp vectors for assembly
847  std::vector<types::global_dof_index> dofs_per_component =
848  DoFTools::count_dofs_per_fe_component(dof_handler);
849  const unsigned int n_sigma = dofs_per_component[0],
850  n_u = dofs_per_component[3];
851 
852  Vector<double> tmp_u(n_u), tmp_sigma(n_sigma);
853 
854  // This assembles the local contribution to the global global matrix
855  // with an algebraic trick. It uses the local system matrix stored in
856  // the respective basis object.
857  unsigned int block_row, block_col;
858 
859  BlockVector<double> *test_vec_ptr, *trial_vec_ptr;
860  unsigned int offset_index = GeometryInfo<3>::vertices_per_cell;
861 
862  for (unsigned int i_test = 0; i_test < length_system_basis; ++i_test)
863  {
864  if (i_test < GeometryInfo<3>::vertices_per_cell)
865  {
866  block_row = 0;
867  test_vec_ptr = &(basis_h1_v[i_test]);
868  }
869  else
870  {
871  block_row = 1;
872  test_vec_ptr = &(basis_curl_v[i_test - offset_index]);
873  }
874 
875  for (unsigned int i_trial = 0; i_trial < length_system_basis; ++i_trial)
876  {
877  if (i_trial < GeometryInfo<3>::vertices_per_cell)
878  {
879  block_col = 0;
880  trial_vec_ptr = &(basis_h1_v[i_trial]);
881  }
882  else
883  {
884  block_col = 1;
885  trial_vec_ptr = &(basis_curl_v[i_trial - offset_index]);
886  }
887 
888  if (block_row == 0) /* This means we are testing with sigma. */
889  {
890  if (block_col == 0) /* This means trial function is sigma. */
891  {
892  assembled_matrix.block(block_row, block_col)
893  .vmult(tmp_sigma, trial_vec_ptr->block(block_col));
894  global_element_matrix(i_test, i_trial) +=
895  (test_vec_ptr->block(block_row) * tmp_sigma);
896  tmp_sigma = 0;
897  }
898  if (block_col == 1) /* This means trial function is u. */
899  {
900  assembled_matrix.block(block_row, block_col)
901  .vmult(tmp_sigma, trial_vec_ptr->block(block_col));
902  global_element_matrix(i_test, i_trial) +=
903  (test_vec_ptr->block(block_row) * tmp_sigma);
904  tmp_sigma = 0;
905  }
906  } // end if
907  else /* This means we are testing with u. */
908  {
909  if (block_col == 0) /* This means trial function is sigma. */
910  {
911  assembled_matrix.block(block_row, block_col)
912  .vmult(tmp_u, trial_vec_ptr->block(block_col));
913  global_element_matrix(i_test, i_trial) +=
914  (test_vec_ptr->block(block_row) * tmp_u);
915  tmp_u = 0;
916  }
917  if (block_col == 1) /* This means trial function is u. */
918  {
919  assembled_matrix.block(block_row, block_col)
920  .vmult(tmp_u, trial_vec_ptr->block(block_col));
921  global_element_matrix(i_test, i_trial) +=
922  test_vec_ptr->block(block_row) * tmp_u;
923  tmp_u = 0;
924  }
925  } // end else
926  } // end for i_trial
927 
928  if (i_test >= GeometryInfo<3>::vertices_per_cell)
929  {
930  block_row = 1;
931  // If we are testing with u we possibly have a
932  // right-hand side.
933  global_element_rhs(i_test) +=
934  test_vec_ptr->block(block_row) * global_rhs.block(block_row);
935  }
936  } // end for i_test
937 
938  is_built_global_element_matrix = true;
939  }
