389  GeometryInfo<3>::faces_per_cell,

396  if (parameters.use_exact_solution)

398  new EquationData::RightHandSideExactLin(parameter_filename));
399  else
400  right_hand_side.reset(
401  new EquationData::RightHandSideParsed(parameter_filename,
402  /* n_components */ 3));
403

405  parameter_filename);

407  parameters.use_exact_solution);

412  std::vector<Tensor<2, 3>> diffusion_inverse_a_values(n_q_points);
413  std::vector<double> diffusion_b_values(n_q_points);
414  std::vector<double> reaction_rate_values(n_q_points);

418  /* degree */ 0);

420  GeometryInfo<3>::lines_per_cell, std::vector<Tensor<1, 3>>(n_q_points));

436  local_rhs = 0;

456  reaction_rate_values);

511  }
512  } // end for ++i
513  } // end for ++q
514

636  void
637  NedRTBasis::solve_iterative(unsigned int n_basis)
638  {
639  Timer timer;
640  Timer inner_timer;
641 
642  // ------------------------------------------
643  // Make a preconditioner for each system matrix
644  if (parameters.verbose)
645  {
646  std::cout << " Computing preconditioner in cell "
647  << global_cell_id.to_string() << " for basis " << n_basis
648  << " .....";
649 
650  timer.restart();
651  }
652 
653  BlockVector<double> *system_rhs_ptr = NULL;
654  BlockVector<double> *solution_ptr = NULL;
655  if (n_basis < GeometryInfo<3>::lines_per_cell)
656  {
657  system_rhs_ptr = &(system_rhs_curl_v[n_basis]);
658  solution_ptr = &(basis_curl_v[n_basis]);
659  }
660  else
661  {
662  const unsigned int offset_index =
663  n_basis - GeometryInfo<3>::lines_per_cell;
664  system_rhs_ptr = &(system_rhs_div_v[offset_index]);
665  solution_ptr = &(basis_div_v[offset_index]);
666  }
667 
668  // for convenience
669  const BlockVector<double> &system_rhs = *system_rhs_ptr;
670  BlockVector<double> & solution = *solution_ptr;
671 
672  inner_schur_preconditioner = std::make_shared<
673  typename LinearSolvers::LocalInnerPreconditioner<3>::type>();
674 
675  typename LinearSolvers::LocalInnerPreconditioner<3>::type::AdditionalData
676  data;
677  inner_schur_preconditioner->initialize(system_matrix.block(0, 0), data);
678 
679  if (parameters.verbose)
680  {
681  timer.stop();
682  std::cout << "done in " << timer.cpu_time() << " seconds."
683  << std::endl;
684  }
685  // ------------------------------------------
686 
687  // Now solve.
688  if (parameters.verbose)
689  {
690  std::cout << " Solving linear system (iteratively, with "
691  "preconditioner) in cell "
692  << global_cell_id.to_string() << " for basis " << n_basis
693  << " .....";
694 
695  timer.restart();
696  }
697 
698  // Construct inverse of upper left block
699  const LinearSolvers::InverseMatrix<
700  SparseMatrix<double>,
701  typename LinearSolvers::LocalInnerPreconditioner<3>::type>
702  block_inverse(system_matrix.block(0, 0), *inner_schur_preconditioner);
703 
704  Vector<double> tmp(system_rhs.block(0).size());
705  {
706  // Set up Schur complement
707  LinearSolvers::SchurComplement<
708  BlockSparseMatrix<double>,
709  Vector<double>,
710  typename LinearSolvers::LocalInnerPreconditioner<3>::type>
711  schur_complement(system_matrix, block_inverse);
712 
713  // Compute schur_rhs = -g + C*A^{-1}*f
714  Vector<double> schur_rhs(system_rhs.block(1).size());
715 
716  block_inverse.vmult(tmp, system_rhs.block(0));
717  system_matrix.block(1, 0).vmult(schur_rhs, tmp);
718  schur_rhs -= system_rhs.block(1);
719 
720  {
721  if (parameters.verbose)
722  {
723  inner_timer.restart();
724  }
725  SolverControl solver_control(system_matrix.m(),
726  1e-6 * schur_rhs.l2_norm());
727  SolverCG<Vector<double>> schur_solver(solver_control);
728 
729  // PreconditionIdentity preconditioner;
730 
731  /*
732  * Precondition the Schur complement with
733  * the approximate inverse of the
734  * Schur complement.
735  */
736  // LinearSolvers::ApproximateInverseMatrix<LinearSolvers::SchurComplement<BlockSparseMatrix<double>,
737  // Vector<double>,
738  // typename
739  // LinearSolvers::LocalInnerPreconditioner<3>::type>,
740  // PreconditionIdentity>
741  // preconditioner
742  //(schur_complement, PreconditionIdentity(),
743  // /* n_iter */
744  // 5);
745 
746  /*
747  * Precondition the Schur complement with
748  * the (approximate) inverse of an approximate
749  * Schur complement.
750  */
751  // using ApproxSchurPrecon =
752  // PreconditionJacobi<SparseMatrix<double>>;
753  // using ApproxSchurPrecon =
754  // PreconditionSOR<SparseMatrix<double>>;
755  using ApproxSchurPrecon = SparseILU<double>;
756  // using ApproxSchurPrecon = PreconditionIdentity;
757  LinearSolvers::ApproximateSchurComplement<BlockSparseMatrix<double>,
758  Vector<double>,
759  ApproxSchurPrecon>
760  approx_schur(system_matrix);
761 
762  LinearSolvers::ApproximateInverseMatrix<
763  LinearSolvers::ApproximateSchurComplement<BlockSparseMatrix<double>,
764  Vector<double>,
765  ApproxSchurPrecon>,
766  PreconditionIdentity>
767  preconditioner(approx_schur,
768  PreconditionIdentity(),
769 #ifdef DEBUG
770  /* n_iter */ 1000);
771 #else
772  /* n_iter */ 14);
773 #endif
774 
775  schur_solver.solve(schur_complement,
776  solution.block(1),
777  schur_rhs,
778  preconditioner);
779 
780  if (n_basis < GeometryInfo<3>::lines_per_cell)
781  {
782  constraints_curl_v[n_basis].distribute(solution);
783  }
784  else
785  {
786  const unsigned int offset_index =
787  n_basis - GeometryInfo<3>::lines_per_cell;
788 
789  constraints_div_v[offset_index].distribute(solution);
790  }
791 
792  if (parameters.verbose)
793  {
794  inner_timer.stop();
795 
796  std::cout << std::endl
797  << " - Iterative Schur complement solver "
798  "converged in "
799  << solver_control.last_step()
800  << " iterations. Time: " << inner_timer.cpu_time()
801  << " seconds." << std::endl;
802  }
803  }
804 
805  {
806  if (parameters.verbose)
807  {
808  inner_timer.restart();
809  }
810 
811  // use computed u to solve for sigma
812  system_matrix.block(0, 1).vmult(tmp, solution.block(1));
813  tmp *= -1;
814  tmp += system_rhs.block(0);
815 
816  // Solve for sigma
817  block_inverse.vmult(solution.block(0), tmp);
818 
819  if (parameters.verbose)
820  {
821  inner_timer.stop();
822 
823  std::cout << " - Outer solver completed. Time: "
824  << inner_timer.cpu_time() << " seconds." << std::endl;
825  }
826  }
827 
828  if (n_basis < GeometryInfo<3>::lines_per_cell)
829  {
830  constraints_curl_v[n_basis].distribute(solution);
831  }
832  else
833  {
834  const unsigned int offset_index =
835  n_basis - GeometryInfo<3>::lines_per_cell;
836 
837  constraints_div_v[offset_index].distribute(solution);
838  }
839  }
840 
841  if (parameters.verbose)
842  {
843  timer.stop();
844  std::cout << " - done in " << timer.cpu_time() << " seconds."
845  << std::endl;
846  }

849  void
850  NedRTBasis::assemble_global_element_matrix()
851  {
852  // First, reset.
853  global_element_matrix = 0;
854 
855  // Get lengths of tmp vectors for assembly
856  std::vector<types::global_dof_index> dofs_per_component =
857  DoFTools::count_dofs_per_fe_component(dof_handler);
858  const unsigned int n_sigma = dofs_per_component[0],
859  n_u = dofs_per_component[3];
860 
861  Vector<double> tmp_u(n_u), tmp_sigma(n_sigma);
862 
863  // This assembles the local contribution to the global global matrix
864  // with an algebraic trick. It uses the local system matrix stored in
865  // the respective basis object.
866  unsigned int block_row, block_col;
867 
868  BlockVector<double> *test_vec_ptr, *trial_vec_ptr;
869  unsigned int offset_index = GeometryInfo<3>::lines_per_cell;
870 
871  for (unsigned int i_test = 0; i_test < length_system_basis; ++i_test)
872  {
873  if (i_test < GeometryInfo<3>::lines_per_cell)
874  {
875  block_row = 0;
876  test_vec_ptr = &(basis_curl_v.at(i_test));
877  }
878  else
879  {
880  block_row = 1;
881  test_vec_ptr = &(basis_div_v.at(i_test - offset_index));
882  }
883 
884  for (unsigned int i_trial = 0; i_trial < length_system_basis; ++i_trial)
885  {
886  if (i_trial < GeometryInfo<3>::lines_per_cell)
887  {
888  block_col = 0;
889  trial_vec_ptr = &(basis_curl_v.at(i_trial));
890  }
891  else
892  {
893  block_col = 1;
894  trial_vec_ptr = &(basis_div_v.at(i_trial - offset_index));
895  }
896 
897  if (block_row == 0) /* This means we are testing with sigma. */
898  {
899  if (block_col == 0) /* This means trial function is sigma. */
900  {
901  assembled_matrix.block(block_row, block_col)
902  .vmult(tmp_sigma, trial_vec_ptr->block(block_col));
903  global_element_matrix(i_test, i_trial) +=
904  (test_vec_ptr->block(block_row) * tmp_sigma);
905  tmp_sigma = 0;
906  }
907  if (block_col == 1) /* This means trial function is u. */
908  {
909  assembled_matrix.block(block_row, block_col)
910  .vmult(tmp_sigma, trial_vec_ptr->block(block_col));
911  global_element_matrix(i_test, i_trial) +=
912  (test_vec_ptr->block(block_row) * tmp_sigma);
913  tmp_sigma = 0;
914  }
915  } // end if
916  else /* This means we are testing with u. */
917  {
918  if (block_col == 0) /* This means trial function is sigma. */
919  {
920  assembled_matrix.block(block_row, block_col)
921  .vmult(tmp_u, trial_vec_ptr->block(block_col));
922  global_element_matrix(i_test, i_trial) +=
923  (test_vec_ptr->block(block_row) * tmp_u);
924  tmp_u = 0;
925  }
926  if (block_col == 1) /* This means trial function is u. */
927  {
928  assembled_matrix.block(block_row, block_col)
929  .vmult(tmp_u, trial_vec_ptr->block(block_col));
930  global_element_matrix(i_test, i_trial) +=
931  test_vec_ptr->block(block_row) * tmp_u;
932  tmp_u = 0;
933  }
934  } // end else
935  } // end for i_trial
936 
937  if (i_test >= GeometryInfo<3>::lines_per_cell)
938  {
939  block_row = 1;
940  // If we are testing with u we possibly have a
941  // right-hand side.
942  global_element_rhs(i_test) +=
943  test_vec_ptr->block(block_row) * global_rhs.block(block_row);
944  }
945  } // end for i_test
946 
947  is_built_global_element_matrix = true;
948  }
