232  // loop over rows
233  for (unsigned int i = 0; i < dofs_per_cell; ++i)

235  // test functions
236  const Tensor<1, 3> tau_i = fe_values[curl].value(i, q);
237  const Tensor<1, 3> curl_tau_i = fe_values[curl].curl(i, q);
238  const double div_v_i = fe_values[flux].divergence(i, q);
239  const Tensor<1, 3> v_i = fe_values[flux].value(i, q);
240 
241  // loop over columns

273  } // end for ++q

276  // conditions

279  for (unsigned int face_n = 0;

282  {
283  if (cell->at_boundary(face_n)

287  // &&
288  // cell->face(face_n)->boundary_id()!=2
289  // /* Select only certain faces. */
290  )
291  {
292  fe_face_values.reinit(cell, face_n);
293 
294  // note that the BVs for u
295  // already have a minus
296  minus_B_div_u->value_list(
297  fe_face_values.get_quadrature_points(),
298  minus_B_div_u_values);
299 
300  u->value_list(fe_face_values.get_quadrature_points(),
301  u_values);
302 
303  for (unsigned int q = 0; q < n_face_q_points; ++q)
304  {
305  const Tensor<1, 3> u_cross_n =
306  cross_product_3d(u_values[q],
307  fe_face_values.normal_vector(q));
308  for (unsigned int i = 0; i < dofs_per_cell; ++i)
309  {
310  local_rhs(i) +=
311  -fe_face_values[flux].value(i, q) *
312  fe_face_values.normal_vector(q) *
313  minus_B_div_u_values[q] *
314  fe_face_values.JxW(q);
315  local_rhs(i) +=
316  -fe_face_values[curl].value(i, q) *
317  u_cross_n * fe_face_values.JxW(q);
318  }
319  }
320  } // end if cell_at_boundary
321  } // end boundary integral
322  } // end if
323

326  constraints.distribute_local_to_global(
327  local_matrix,
328  local_rhs,
329  local_dof_indices,
330  system_matrix,
331  system_rhs,
332  /* use inhomogeneities for rhs */ true);
333  }
334  } // end for ++cell

337  system_rhs.compress(VectorOperation::add);
